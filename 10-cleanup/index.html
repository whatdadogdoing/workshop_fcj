Module 10: Clean Up Resources

Overview

Clean up all AWS resources created during this workshop to avoid ongoing charges and maintain good cloud hygiene practices. This module provides comprehensive cleanup procedures and automation scripts.

Learning Objectives

  • Verify complete resource cleanup

Resource Inventory

# Python script to inventory all workshop resources
import boto3
import json
from datetime import datetime

class ResourceInventory:
    def __init__(self):
        self.ecs = boto3.client('ecs')
        self.ecr = boto3.client('ecr')
        self.rds = boto3.client('rds')
        self.elbv2 = boto3.client('elbv2')
        self.ec2 = boto3.client('ec2')
        self.logs = boto3.client('logs')
        self.iam = boto3.client('iam')
        self.s3 = boto3.client('s3')
        self.codedeploy = boto3.client('codedeploy')
        self.cloudwatch = boto3.client('cloudwatch')
        
    def get_tagged_resources(self, tag_key='Project', tag_value='enterprise-app'):
        """Get all resources with specific tags"""
        resource_groups = boto3.client('resourcegroupstaggingapi')
        
        response = resource_groups.get_resources(
            TagFilters=[
                {
                    'Key': tag_key,
                    'Values': [tag_value]
                }
            ]
        )
        
        return response.get('ResourceTagMappingList', [])
    
    def inventory_ecs_resources(self):
        """Inventory ECS clusters, services, and tasks"""
        resources = []
        
        # List clusters
        clusters = self.ecs.list_clusters()['clusterArns']
        for cluster_arn in clusters:
            if 'enterprise' in cluster_arn:
                cluster_name = cluster_arn.split('/')[-1]
                resources.append({
                    'type': 'ECS Cluster',
                    'name': cluster_name,
                    'arn': cluster_arn
                })
                
                # List services in cluster
                services = self.ecs.list_services(cluster=cluster_arn)['serviceArns']
                for service_arn in services:
                    service_name = service_arn.split('/')[-1]
                    resources.append({
                        'type': 'ECS Service',
                        'name': service_name,
                        'arn': service_arn,
                        'cluster': cluster_name
                    })
        
        return resources
    
    def inventory_ecr_repositories(self):
        """Inventory ECR repositories"""
        resources = []
        
        repositories = self.ecr.describe_repositories()['repositories']
        for repo in repositories:
            if 'enterprise' in repo['repositoryName']:
                resources.append({
                    'type': 'ECR Repository',
                    'name': repo['repositoryName'],
                    'arn': repo['repositoryArn'],
                    'uri': repo['repositoryUri']
                })
        
        return resources
    
    def inventory_rds_resources(self):
        """Inventory RDS instances and snapshots"""
        resources = []
        
        # DB Instances
        instances = self.rds.describe_db_instances()['DBInstances']
        for instance in instances:
            if 'enterprise' in instance['DBInstanceIdentifier']:
                resources.append({
                    'type': 'RDS Instance',
                    'name': instance['DBInstanceIdentifier'],
                    'arn': instance['DBInstanceArn'],
                    'status': instance['DBInstanceStatus']
                })
        
        # DB Snapshots
        snapshots = self.rds.describe_db_snapshots()['DBSnapshots']
        for snapshot in snapshots:
            if 'enterprise' in snapshot['DBSnapshotIdentifier']:
                resources.append({
                    'type': 'RDS Snapshot',
                    'name': snapshot['DBSnapshotIdentifier'],
                    'arn': snapshot['DBSnapshotArn'],
                    'status': snapshot['Status']
                })
        
        return resources
    
    def inventory_load_balancers(self):
        """Inventory Application Load Balancers"""
        resources = []
        
        load_balancers = self.elbv2.describe_load_balancers()['LoadBalancers']
        for lb in load_balancers:
            if 'enterprise' in lb['LoadBalancerName']:
                resources.append({
                    'type': 'Application Load Balancer',
                    'name': lb['LoadBalancerName'],
                    'arn': lb['LoadBalancerArn'],
                    'dns_name': lb['DNSName']
                })
                
                # Get target groups
                target_groups = self.elbv2.describe_target_groups(
                    LoadBalancerArn=lb['LoadBalancerArn']
                )['TargetGroups']
                
                for tg in target_groups:
                    resources.append({
                        'type': 'Target Group',
                        'name': tg['TargetGroupName'],
                        'arn': tg['TargetGroupArn'],
                        'load_balancer': lb['LoadBalancerName']
                    })
        
        return resources
    
    def generate_inventory_report(self):
        """Generate complete resource inventory"""
        inventory = {
            'timestamp': datetime.utcnow().isoformat(),
            'resources': {
                'ecs': self.inventory_ecs_resources(),
                'ecr': self.inventory_ecr_repositories(),
                'rds': self.inventory_rds_resources(),
                'load_balancers': self.inventory_load_balancers()
            }
        }
        
        # Add tagged resources
        inventory['tagged_resources'] = self.get_tagged_resources()
        
        # Calculate total resource count
        total_resources = sum(len(resources) for resources in inventory['resources'].values())
        inventory['total_resources'] = total_resources
        
        return inventory

# Usage
inventory = ResourceInventory()
report = inventory.generate_inventory_report()

print(f"Total resources found: {report['total_resources']}")
print(json.dumps(report, indent=2, default=str))

Systematic Cleanup Procedure

Step 1: ECS Resources

# Complete ECS cleanup script
#!/bin/bash

set -e

echo "Starting ECS resource cleanup..."

# Function to wait for service to scale down
wait_for_service_stable() {
    local cluster=$1
    local service=$2
    
    echo "Waiting for service $service to become stable..."
    aws ecs wait services-stable --cluster $cluster --services $service
}

# Function to delete ECS service
delete_ecs_service() {
    local cluster=$1
    local service=$2
    
    echo "Scaling down service: $service"
    aws ecs update-service \
        --cluster $cluster \
        --service $service \
        --desired-count 0
    
    wait_for_service_stable $cluster $service
    
    echo "Deleting service: $service"
    aws ecs delete-service \
        --cluster $cluster \
        --service $service
    
    echo "Waiting for service deletion..."
    aws ecs wait services-inactive --cluster $cluster --services $service
}

# Get all enterprise clusters
CLUSTERS=$(aws ecs list-clusters --query 'clusterArns[?contains(@, `enterprise`)]' --output text)

for CLUSTER_ARN in $CLUSTERS; do
    CLUSTER_NAME=$(echo $CLUSTER_ARN | cut -d'/' -f2)
    echo "Processing cluster: $CLUSTER_NAME"
    
    # Get all services in cluster
    SERVICES=$(aws ecs list-services --cluster $CLUSTER_ARN --query 'serviceArns' --output text)
    
    for SERVICE_ARN in $SERVICES; do
        SERVICE_NAME=$(echo $SERVICE_ARN | cut -d'/' -f3)
        delete_ecs_service $CLUSTER_NAME $SERVICE_NAME
    done
    
    # Delete task definitions
    TASK_DEFINITIONS=$(aws ecs list-task-definitions --family-prefix enterprise --query 'taskDefinitionArns' --output text)
    
    for TASK_DEF_ARN in $TASK_DEFINITIONS; do
        echo "Deregistering task definition: $TASK_DEF_ARN"
        aws ecs deregister-task-definition --task-definition $TASK_DEF_ARN
    done
    
    # Delete cluster
    echo "Deleting cluster: $CLUSTER_NAME"
    aws ecs delete-cluster --cluster $CLUSTER_ARN
done

echo "ECS cleanup completed!"

Step 2: Load Balancer Cleanup

# Load balancer cleanup script
#!/bin/bash

set -e

echo "Starting Load Balancer cleanup..."

# Get all enterprise load balancers
LOAD_BALANCERS=$(aws elbv2 describe-load-balancers \
    --query 'LoadBalancers[?contains(LoadBalancerName, `enterprise`)].LoadBalancerArn' \
    --output text)

for LB_ARN in $LOAD_BALANCERS; do
    LB_NAME=$(aws elbv2 describe-load-balancers \
        --load-balancer-arns $LB_ARN \
        --query 'LoadBalancers[0].LoadBalancerName' \
        --output text)
    
    echo "Processing load balancer: $LB_NAME"
    
    # Get listeners
    LISTENERS=$(aws elbv2 describe-listeners \
        --load-balancer-arn $LB_ARN \
        --query 'Listeners[].ListenerArn' \
        --output text)
    
    # Delete listeners
    for LISTENER_ARN in $LISTENERS; do
        echo "Deleting listener: $LISTENER_ARN"
        aws elbv2 delete-listener --listener-arn $LISTENER_ARN
    done
    
    # Get target groups
    TARGET_GROUPS=$(aws elbv2 describe-target-groups \
        --load-balancer-arn $LB_ARN \
        --query 'TargetGroups[].TargetGroupArn' \
        --output text)
    
    # Delete load balancer
    echo "Deleting load balancer: $LB_NAME"
    aws elbv2 delete-load-balancer --load-balancer-arn $LB_ARN
    
    # Wait for load balancer deletion
    echo "Waiting for load balancer deletion..."
    aws elbv2 wait load-balancer-not-exists --load-balancer-arns $LB_ARN
    
    # Delete target groups
    for TG_ARN in $TARGET_GROUPS; do
        echo "Deleting target group: $TG_ARN"
        aws elbv2 delete-target-group --target-group-arn $TG_ARN || true
    done
done

echo "Load Balancer cleanup completed!"

Step 3: RDS Cleanup

# RDS cleanup script
#!/bin/bash

set -e

echo "Starting RDS cleanup..."

# Get all enterprise RDS instances
DB_INSTANCES=$(aws rds describe-db-instances \
    --query 'DBInstances[?contains(DBInstanceIdentifier, `enterprise`)].DBInstanceIdentifier' \
    --output text)

for DB_INSTANCE in $DB_INSTANCES; do
    echo "Processing RDS instance: $DB_INSTANCE"
    
    # Check if instance exists and get status
    DB_STATUS=$(aws rds describe-db-instances \
        --db-instance-identifier $DB_INSTANCE \
        --query 'DBInstances[0].DBInstanceStatus' \
        --output text 2>/dev/null || echo "not-found")
    
    if [ "$DB_STATUS" != "not-found" ]; then
        echo "Current status: $DB_STATUS"
        
        # Create final snapshot before deletion
        SNAPSHOT_ID="${DB_INSTANCE}-final-snapshot-$(date +%Y%m%d%H%M%S)"
        echo "Creating final snapshot: $SNAPSHOT_ID"
        
        aws rds create-db-snapshot \
            --db-instance-identifier $DB_INSTANCE \
            --db-snapshot-identifier $SNAPSHOT_ID
        
        # Wait for snapshot to complete
        echo "Waiting for snapshot to complete..."
        aws rds wait db-snapshot-completed --db-snapshot-identifier $SNAPSHOT_ID
        
        # Delete RDS instance
        echo "Deleting RDS instance: $DB_INSTANCE"
        aws rds delete-db-instance \
            --db-instance-identifier $DB_INSTANCE \
            --skip-final-snapshot \
            --delete-automated-backups
        
        # Wait for deletion
        echo "Waiting for RDS instance deletion..."
        aws rds wait db-instance-deleted --db-instance-identifier $DB_INSTANCE
    fi
done

# Clean up old snapshots (keep only final snapshots)
echo "Cleaning up old snapshots..."
OLD_SNAPSHOTS=$(aws rds describe-db-snapshots \
    --query 'DBSnapshots[?contains(DBSnapshotIdentifier, `enterprise`) && !contains(DBSnapshotIdentifier, `final-snapshot`)].DBSnapshotIdentifier' \
    --output text)

for SNAPSHOT in $OLD_SNAPSHOTS; do
    echo "Deleting old snapshot: $SNAPSHOT"
    aws rds delete-db-snapshot --db-snapshot-identifier $SNAPSHOT || true
done

echo "RDS cleanup completed!"

Step 4: ECR Cleanup

# ECR cleanup script
#!/bin/bash

set -e

echo "Starting ECR cleanup..."

# Get all enterprise ECR repositories
REPOSITORIES=$(aws ecr describe-repositories \
    --query 'repositories[?contains(repositoryName, `enterprise`)].repositoryName' \
    --output text)

for REPO in $REPOSITORIES; do
    echo "Processing ECR repository: $REPO"
    
    # List all images in repository
    IMAGES=$(aws ecr list-images --repository-name $REPO --query 'imageIds' --output json)
    
    if [ "$IMAGES" != "[]" ]; then
        echo "Deleting all images in repository: $REPO"
        aws ecr batch-delete-image \
            --repository-name $REPO \
            --image-ids "$IMAGES"
    fi
    
    # Delete repository
    echo "Deleting ECR repository: $REPO"
    aws ecr delete-repository \
        --repository-name $REPO \
        --force
done

echo "ECR cleanup completed!"

# Additional resources cleanup script
#!/bin/bash

set -e

echo "Starting additional resources cleanup..."

# CloudWatch Log Groups
echo "Deleting CloudWatch Log Groups..."
LOG_GROUPS=$(aws logs describe-log-groups \
    --query 'logGroups[?contains(logGroupName, `enterprise`)].logGroupName' \
    --output text)

for LOG_GROUP in $LOG_GROUPS; do
    echo "Deleting log group: $LOG_GROUP"
    aws logs delete-log-group --log-group-name "$LOG_GROUP"
done

# CloudWatch Alarms
echo "Deleting CloudWatch Alarms..."
ALARMS=$(aws cloudwatch describe-alarms \
    --query 'MetricAlarms[?contains(AlarmName, `enterprise`) || contains(AlarmName, `ECS`) || contains(AlarmName, `ALB`)].AlarmName' \
    --output text)

for ALARM in $ALARMS; do
    echo "Deleting alarm: $ALARM"
    aws cloudwatch delete-alarms --alarm-names "$ALARM"
done

# CodeDeploy Applications
echo "Deleting CodeDeploy Applications..."
APPLICATIONS=$(aws deploy list-applications \
    --query 'applications[?contains(@, `enterprise`)]' \
    --output text)

for APP in $APPLICATIONS; do
    echo "Deleting CodeDeploy application: $APP"
    aws deploy delete-application --application-name "$APP"
done

# S3 Buckets (be careful with this!)
echo "Listing S3 buckets for manual review..."
BUCKETS=$(aws s3api list-buckets \
    --query 'Buckets[?contains(Name, `enterprise`)].Name' \
    --output text)

for BUCKET in $BUCKETS; do
    echo "Found S3 bucket: $BUCKET (manual deletion required)"
    echo "To delete: aws s3 rb s3://$BUCKET --force"
done

# IAM Roles and Policies
echo "Listing IAM roles for manual review..."
ROLES=$(aws iam list-roles \
    --query 'Roles[?contains(RoleName, `enterprise`) || contains(RoleName, `ecs`) || contains(RoleName, `codedeploy`)].RoleName' \
    --output text)

for ROLE in $ROLES; do
    echo "Found IAM role: $ROLE (manual deletion required)"
done

echo "Additional resources cleanup completed!"
echo "Please manually review and delete S3 buckets and IAM roles as needed."

Automated Cleanup Script

Master Cleanup Script

# Master cleanup script
#!/bin/bash

set -e

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

echo -e "${YELLOW}Enterprise CI/CD Workshop Cleanup Script${NC}"
echo -e "${YELLOW}=========================================${NC}"
echo

# Confirmation prompt
read -p "This will delete ALL workshop resources. Are you sure? (yes/no): " -r
echo
if [[ ! $REPLY =~ ^[Yy][Ee][Ss]$ ]]; then
    echo "Cleanup cancelled."
    exit 1
fi

# Set AWS region
export AWS_DEFAULT_REGION=${AWS_REGION:-us-east-1}
echo -e "${GREEN}Using AWS region: $AWS_DEFAULT_REGION${NC}"

# Function to check if AWS CLI is configured
check_aws_config() {
    if ! aws sts get-caller-identity >/dev/null 2>&1; then
        echo -e "${RED}Error: AWS CLI not configured or no valid credentials${NC}"
        exit 1
    fi
    
    ACCOUNT_ID=$(aws sts get-caller-identity --query Account --output text)
    echo -e "${GREEN}AWS Account ID: $ACCOUNT_ID${NC}"
}

# Function to log cleanup steps
log_step() {
    echo -e "${YELLOW}[$(date '+%Y-%m-%d %H:%M:%S')] $1${NC}"
}

# Function to handle errors
handle_error() {
    echo -e "${RED}Error in step: $1${NC}"
    echo -e "${RED}$2${NC}"
    read -p "Continue with cleanup? (yes/no): " -r
    if [[ ! $REPLY =~ ^[Yy][Ee][Ss]$ ]]; then
        exit 1
    fi
}

# Check AWS configuration
check_aws_config

# Step 1: ECS Resources
log_step "Cleaning up ECS resources..."
if ! ./cleanup-ecs.sh; then
    handle_error "ECS cleanup" "Failed to clean up ECS resources"
fi

# Step 2: Load Balancers
log_step "Cleaning up Load Balancers..."
if ! ./cleanup-alb.sh; then
    handle_error "ALB cleanup" "Failed to clean up Load Balancers"
fi

# Step 3: RDS
log_step "Cleaning up RDS resources..."
if ! ./cleanup-rds.sh; then
    handle_error "RDS cleanup" "Failed to clean up RDS resources"
fi

# Step 4: ECR
log_step "Cleaning up ECR repositories..."
if ! ./cleanup-ecr.sh; then
    handle_error "ECR cleanup" "Failed to clean up ECR repositories"
fi

# Step 5: Additional Resources
log_step "Cleaning up additional resources..."
if ! ./cleanup-additional.sh; then
    handle_error "Additional cleanup" "Failed to clean up additional resources"
fi

# Final verification
log_step "Running final verification..."
python3 verify-cleanup.py

echo
echo -e "${GREEN}=========================================${NC}"
echo -e "${GREEN}Cleanup completed successfully!${NC}"
echo -e "${GREEN}=========================================${NC}"
echo
echo -e "${YELLOW}Please manually review and delete:${NC}"
echo "- S3 buckets (if any)"
echo "- IAM roles and policies (if any)"
echo "- Route 53 hosted zones (if any)"
echo "- CloudFormation stacks (if any)"
echo
echo -e "${YELLOW}Check your AWS billing dashboard to confirm no ongoing charges.${NC}"

Verification Script

# Python verification script
import boto3
import sys
from datetime import datetime

class CleanupVerifier:
    def __init__(self):
        self.ecs = boto3.client('ecs')
        self.ecr = boto3.client('ecr')
        self.rds = boto3.client('rds')
        self.elbv2 = boto3.client('elbv2')
        self.logs = boto3.client('logs')
        self.cloudwatch = boto3.client('cloudwatch')
        self.codedeploy = boto3.client('codedeploy')
        
        self.errors = []
        self.warnings = []
    
    def verify_ecs_cleanup(self):
        """Verify ECS resources are deleted"""
        print("Verifying ECS cleanup...")
        
        # Check clusters
        clusters = self.ecs.list_clusters()['clusterArns']
        enterprise_clusters = [c for c in clusters if 'enterprise' in c]
        
        if enterprise_clusters:
            self.errors.append(f"Found {len(enterprise_clusters)} ECS clusters still exist")
            for cluster in enterprise_clusters:
                print(f"  - {cluster}")
        else:
            print("  ✓ No ECS clusters found")
        
        # Check task definitions
        task_defs = self.ecs.list_task_definitions(familyPrefix='enterprise')['taskDefinitionArns']
        active_task_defs = [td for td in task_defs if ':ACTIVE' in td or td.endswith(':1')]
        
        if active_task_defs:
            self.warnings.append(f"Found {len(active_task_defs)} active task definitions")
        else:
            print("  ✓ No active task definitions found")
    
    def verify_ecr_cleanup(self):
        """Verify ECR repositories are deleted"""
        print("Verifying ECR cleanup...")
        
        try:
            repositories = self.ecr.describe_repositories()['repositories']
            enterprise_repos = [r for r in repositories if 'enterprise' in r['repositoryName']]
            
            if enterprise_repos:
                self.errors.append(f"Found {len(enterprise_repos)} ECR repositories still exist")
                for repo in enterprise_repos:
                    print(f"  - {repo['repositoryName']}")
            else:
                print("  ✓ No ECR repositories found")
        except Exception as e:
            print(f"  ✓ ECR verification completed (no repositories)")
    
    def verify_rds_cleanup(self):
        """Verify RDS resources are deleted"""
        print("Verifying RDS cleanup...")
        
        # Check DB instances
        instances = self.rds.describe_db_instances()['DBInstances']
        enterprise_instances = [i for i in instances if 'enterprise' in i['DBInstanceIdentifier']]
        
        if enterprise_instances:
            self.errors.append(f"Found {len(enterprise_instances)} RDS instances still exist")
            for instance in enterprise_instances:
                print(f"  - {instance['DBInstanceIdentifier']} ({instance['DBInstanceStatus']})")
        else:
            print("  ✓ No RDS instances found")
        
        # Check snapshots
        snapshots = self.rds.describe_db_snapshots()['DBSnapshots']
        enterprise_snapshots = [s for s in snapshots if 'enterprise' in s['DBSnapshotIdentifier']]
        
        if enterprise_snapshots:
            final_snapshots = [s for s in enterprise_snapshots if 'final-snapshot' in s['DBSnapshotIdentifier']]
            other_snapshots = [s for s in enterprise_snapshots if 'final-snapshot' not in s['DBSnapshotIdentifier']]
            
            if other_snapshots:
                self.warnings.append(f"Found {len(other_snapshots)} non-final snapshots")
            
            if final_snapshots:
                print(f"  ✓ Found {len(final_snapshots)} final snapshots (expected)")
        else:
            print("  ✓ No RDS snapshots found")
    
    def verify_load_balancer_cleanup(self):
        """Verify Load Balancer resources are deleted"""
        print("Verifying Load Balancer cleanup...")
        
        load_balancers = self.elbv2.describe_load_balancers()['LoadBalancers']
        enterprise_lbs = [lb for lb in load_balancers if 'enterprise' in lb['LoadBalancerName']]
        
        if enterprise_lbs:
            self.errors.append(f"Found {len(enterprise_lbs)} Load Balancers still exist")
            for lb in enterprise_lbs:
                print(f"  - {lb['LoadBalancerName']}")
        else:
            print("  ✓ No Load Balancers found")
        
        # Check target groups
        target_groups = self.elbv2.describe_target_groups()['TargetGroups']
        enterprise_tgs = [tg for tg in target_groups if 'enterprise' in tg['TargetGroupName']]
        
        if enterprise_tgs:
            self.errors.append(f"Found {len(enterprise_tgs)} Target Groups still exist")
        else:
            print("  ✓ No Target Groups found")
    
    def verify_cloudwatch_cleanup(self):
        """Verify CloudWatch resources are cleaned up"""
        print("Verifying CloudWatch cleanup...")
        
        # Check log groups
        try:
            log_groups = self.logs.describe_log_groups()['logGroups']
            enterprise_logs = [lg for lg in log_groups if 'enterprise' in lg['logGroupName']]
            
            if enterprise_logs:
                self.warnings.append(f"Found {len(enterprise_logs)} CloudWatch Log Groups")
            else:
                print("  ✓ No CloudWatch Log Groups found")
        except Exception:
            print("  ✓ CloudWatch Logs verification completed")
        
        # Check alarms
        try:
            alarms = self.cloudwatch.describe_alarms()['MetricAlarms']
            enterprise_alarms = [a for a in alarms if 'enterprise' in a['AlarmName'].lower() or 
                               'ecs' in a['AlarmName'].lower() or 'alb' in a['AlarmName'].lower()]
            
            if enterprise_alarms:
                self.warnings.append(f"Found {len(enterprise_alarms)} CloudWatch Alarms")
            else:
                print("  ✓ No CloudWatch Alarms found")
        except Exception:
            print("  ✓ CloudWatch Alarms verification completed")
    
    def verify_codedeploy_cleanup(self):
        """Verify CodeDeploy resources are cleaned up"""
        print("Verifying CodeDeploy cleanup...")
        
        try:
            applications = self.codedeploy.list_applications()['applications']
            enterprise_apps = [app for app in applications if 'enterprise' in app]
            
            if enterprise_apps:
                self.warnings.append(f"Found {len(enterprise_apps)} CodeDeploy Applications")
            else:
                print("  ✓ No CodeDeploy Applications found")
        except Exception:
            print("  ✓ CodeDeploy verification completed")
    
    def run_verification(self):
        """Run complete cleanup verification"""
        print("\n" + "="*50)
        print("CLEANUP VERIFICATION REPORT")
        print("="*50)
        print(f"Timestamp: {datetime.utcnow().isoformat()}")
        print()
        
        self.verify_ecs_cleanup()
        self.verify_ecr_cleanup()
        self.verify_rds_cleanup()
        self.verify_load_balancer_cleanup()
        self.verify_cloudwatch_cleanup()
        self.verify_codedeploy_cleanup()
        
        print("\n" + "="*50)
        print("VERIFICATION SUMMARY")
        print("="*50)
        
        if self.errors:
            print(f"❌ ERRORS ({len(self.errors)}):")
            for error in self.errors:
                print(f"   - {error}")
        
        if self.warnings:
            print(f"⚠️  WARNINGS ({len(self.warnings)}):")
            for warning in self.warnings:
                print(f"   - {warning}")
        
        if not self.errors and not self.warnings:
            print("✅ All resources successfully cleaned up!")
        elif not self.errors:
            print("✅ Cleanup completed with warnings (manual review recommended)")
        else:
            print("❌ Cleanup incomplete - errors found")
            return False
        
        print("\n💡 Don't forget to check your AWS billing dashboard!")
        return True

if __name__ == "__main__":
    verifier = CleanupVerifier()
