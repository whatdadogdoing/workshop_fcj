Module 8: Deployment Strategies

Overview

Implement advanced deployment strategies including blue-green, canary, and rolling deployments for zero-downtime releases using AWS ECS, CodeDeploy, and Application Load Balancer.

Learning Objectives

  • Implement blue-green deployments with AWS CodeDeploy
  • Configure canary deployments for gradual rollouts
  • Set up rolling deployments with ECS
  • Create automated rollback mechanisms
  • Implement feature flags for controlled releases
  • Monitor deployment health and performance

GitHub Actions Blue-Green Pipeline

# .github/workflows/blue-green-deploy.yml
name: Blue-Green Deployment

on:
  push:
    branches: [main]
  workflow_dispatch:
    inputs:
      environment:
        description: 'Deployment environment'
        required: true
        default: 'production'
        type: choice
        options:
          - staging
          - production

env:
  AWS_REGION: us-east-1
  ECR_REPOSITORY: enterprise-app
  ECS_SERVICE: enterprise-service
  ECS_CLUSTER: enterprise-cluster
  CODEDEPLOY_APPLICATION: enterprise-app
  CODEDEPLOY_DEPLOYMENT_GROUP: production

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v2
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Login to Amazon ECR
        id: login-ecr
        uses: aws-actions/amazon-ecr-login@v1

      - name: Build and push Docker image
        env:
          ECR_REGISTRY: ${{ steps.login-ecr.outputs.registry }}
          IMAGE_TAG: ${{ github.sha }}
        run: |
          docker build -t $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG ./backend
          docker push $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG
          echo "image=$ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG" >> $GITHUB_OUTPUT

      - name: Update ECS task definition
        id: task-def
        uses: aws-actions/amazon-ecs-render-task-definition@v1
        with:
          task-definition: task-definition.json
          container-name: backend
          image: ${{ steps.build-image.outputs.image }}

      - name: Create CodeDeploy deployment
        run: |
          aws deploy create-deployment \
            --application-name $CODEDEPLOY_APPLICATION \
            --deployment-group-name $CODEDEPLOY_DEPLOYMENT_GROUP \
            --revision revisionType=S3,s3Location='{"bucket":"enterprise-deployments","key":"appspec.yml","bundleType":"YAML"}' \
            --deployment-config-name CodeDeployDefault.ECSBlueGreenCanary10Percent5Minutes \
            --description "Blue-Green deployment from GitHub Actions"

      - name: Wait for deployment completion
        run: |
          DEPLOYMENT_ID=$(aws deploy list-deployments \
            --application-name $CODEDEPLOY_APPLICATION \
            --deployment-group-name $CODEDEPLOY_DEPLOYMENT_GROUP \
            --include-only-statuses InProgress \
            --query 'deployments[0]' --output text)
          
          aws deploy wait deployment-successful --deployment-id $DEPLOYMENT_ID
          
          echo "Deployment completed successfully!"

Automated Canary Analysis

# Python script for canary analysis
import boto3
import time
import json
from datetime import datetime, timedelta

class CanaryAnalyzer:
    def __init__(self):
        self.cloudwatch = boto3.client('cloudwatch')
        self.codedeploy = boto3.client('codedeploy')
        
    def get_metric_data(self, metric_name, namespace, dimensions, start_time, end_time):
        response = self.cloudwatch.get_metric_statistics(
            Namespace=namespace,
            MetricName=metric_name,
            Dimensions=dimensions,
            StartTime=start_time,
            EndTime=end_time,
            Period=300,
            Statistics=['Average']
        )
        return response['Datapoints']
    
    def analyze_canary_health(self, deployment_id):
        end_time = datetime.utcnow()
        start_time = end_time - timedelta(minutes=10)
        
        # Get error rate for both versions
        blue_errors = self.get_metric_data(
            'HTTPCode_Target_5XX_Count',
            'AWS/ApplicationELB',
            [{'Name': 'TargetGroup', 'Value': 'enterprise-blue'}],
            start_time, end_time
        )
        
        green_errors = self.get_metric_data(
            'HTTPCode_Target_5XX_Count',
            'AWS/ApplicationELB',
            [{'Name': 'TargetGroup', 'Value': 'enterprise-green'}],
            start_time, end_time
        )
        
        # Get response time for both versions
        blue_response_time = self.get_metric_data(
            'TargetResponseTime',
            'AWS/ApplicationELB',
            [{'Name': 'TargetGroup', 'Value': 'enterprise-blue'}],
            start_time, end_time
        )
        
        green_response_time = self.get_metric_data(
            'TargetResponseTime',
            'AWS/ApplicationELB',
            [{'Name': 'TargetGroup', 'Value': 'enterprise-green'}],
            start_time, end_time
        )
        
        # Analyze metrics
        blue_avg_errors = sum(d['Average'] for d in blue_errors) / len(blue_errors) if blue_errors else 0
        green_avg_errors = sum(d['Average'] for d in green_errors) / len(green_errors) if green_errors else 0
        
        blue_avg_response = sum(d['Average'] for d in blue_response_time) / len(blue_response_time) if blue_response_time else 0
        green_avg_response = sum(d['Average'] for d in green_response_time) / len(green_response_time) if green_response_time else 0
        
        # Decision logic
        error_threshold = 5  # 5% error rate threshold
        response_time_threshold = 2  # 2 second response time threshold
        
        if green_avg_errors > blue_avg_errors * 1.5 or green_avg_errors > error_threshold:
            return self.stop_deployment(deployment_id, "High error rate detected")
        
        if green_avg_response > blue_avg_response * 1.5 or green_avg_response > response_time_threshold:
            return self.stop_deployment(deployment_id, "High response time detected")
        
        return {"status": "healthy", "continue": True}
    
    def stop_deployment(self, deployment_id, reason):
        self.codedeploy.stop_deployment(
            deploymentId=deployment_id,
            autoRollbackEnabled=True
        )
        return {"status": "stopped", "reason": reason, "continue": False}
    
    def continue_deployment(self, deployment_id):
        # Increase canary traffic percentage
        return self.codedeploy.continue_deployment(
            deploymentId=deployment_id,
            deploymentWaitType='READY_WAIT'
        )

# Lambda function for automated canary analysis
def lambda_handler(event, context):
    analyzer = CanaryAnalyzer()
    deployment_id = event['deployment_id']
    
    result = analyzer.analyze_canary_health(deployment_id)
    
    if result['continue']:
        analyzer.continue_deployment(deployment_id)
    
    return {
        'statusCode': 200,
        'body': json.dumps(result)
    }

Feature Flag Implementation

# Python feature flag service
import redis
import json
from typing import Dict, Any, Optional

class FeatureFlagService:
    def __init__(self, redis_url: str):
        self.redis_client = redis.from_url(redis_url)
        self.cache_ttl = 300  # 5 minutes
    
    def is_enabled(self, flag_name: str, user_id: Optional[str] = None, 
                   context: Optional[Dict[str, Any]] = None) -> bool:
        """Check if a feature flag is enabled for a user/context"""
        
        # Get flag configuration
        flag_config = self.get_flag_config(flag_name)
        if not flag_config:
            return False
        
        # Check if flag is globally enabled
        if not flag_config.get('enabled', False):
            return False
        
        # Check percentage rollout
        rollout_percentage = flag_config.get('rollout_percentage', 0)
        if rollout_percentage == 100:
            return True
        elif rollout_percentage == 0:
            return False
        
        # Use user_id for consistent rollout
        if user_id:
            user_hash = hash(f"{flag_name}:{user_id}") % 100
            return user_hash < rollout_percentage
        
        # Check targeting rules
        targeting_rules = flag_config.get('targeting_rules', [])
        for rule in targeting_rules:
            if self.evaluate_rule(rule, context):
                return rule.get('enabled', False)
        
        return False
    
    def get_flag_config(self, flag_name: str) -> Optional[Dict[str, Any]]:
        """Get feature flag configuration from Redis"""
        cache_key = f"feature_flag:{flag_name}"
        
        # Try cache first
        cached_config = self.redis_client.get(cache_key)
        if cached_config:
            return json.loads(cached_config)
        
        # Fallback to database or config service
        config = self.fetch_flag_from_database(flag_name)
        if config:
            self.redis_client.setex(cache_key, self.cache_ttl, json.dumps(config))
        
        return config
    
    def evaluate_rule(self, rule: Dict[str, Any], context: Optional[Dict[str, Any]]) -> bool:
        """Evaluate targeting rule against context"""
        if not context:
            return False
        
        conditions = rule.get('conditions', [])
        for condition in conditions:
            attribute = condition.get('attribute')
            operator = condition.get('operator')
            value = condition.get('value')
            
            context_value = context.get(attribute)
            
            if operator == 'equals' and context_value != value:
                return False
            elif operator == 'in' and context_value not in value:
                return False
            elif operator == 'greater_than' and context_value <= value:
                return False
        
        return True
    
    def fetch_flag_from_database(self, flag_name: str) -> Optional[Dict[str, Any]]:
        """Fetch flag configuration from database"""

# CLI tool for managing feature flags
#!/usr/bin/env python3
import click
import redis
import json

@click.group()
def cli():
    pass

@cli.command()
@click.argument('flag_name')
@click.option('--percentage', default=0, help='Rollout percentage (0-100)')
@click.option('--enabled/--disabled', default=True, help='Enable or disable flag')
def create_flag(flag_name, percentage, enabled):
    """Create a new feature flag"""
    r = redis.from_url('redis://localhost:6379')
    
    flag_config = {
        'enabled': enabled,
        'rollout_percentage': percentage,
        'targeting_rules': [],
        'created_at': datetime.utcnow().isoformat()
    }
    
    r.setex(f"feature_flag:{flag_name}", 86400, json.dumps(flag_config))
    click.echo(f"Created feature flag: {flag_name}")

@cli.command()
@click.argument('flag_name')
@click.argument('percentage', type=int)
def set_rollout(flag_name, percentage):
    """Set rollout percentage for a feature flag"""
    r = redis.from_url('redis://localhost:6379')
    
    flag_config = json.loads(r.get(f"feature_flag:{flag_name}") or '{}')
    flag_config['rollout_percentage'] = percentage
    flag_config['updated_at'] = datetime.utcnow().isoformat()
    
    r.setex(f"feature_flag:{flag_name}", 86400, json.dumps(flag_config))
    click.echo(f"Set rollout percentage for {flag_name} to {percentage}%")

if __name__ == '__main__':
    cli()

    health_status = {
        'status': 'healthy',
        'timestamp': datetime.utcnow().isoformat(),

        'checks': {}
    }
    
    # Database connectivity check
    try:
        db.session.execute('SELECT 1')
        health_status['checks']['database'] = 'healthy'
    except Exception as e:
        health_status['checks']['database'] = f'unhealthy: {str(e)}'
        health_status['status'] = 'unhealthy'
    
    # Redis connectivity check
    try:
        redis_client.ping()
        health_status['checks']['redis'] = 'healthy'
    except Exception as e:
        health_status['checks']['redis'] = f'unhealthy: {str(e)}'
        health_status['status'] = 'unhealthy'
    
    # External service check
    try:
        response = requests.get('https://api.external-service.com/health', timeout=5)
        if response.status_code == 200:
            health_status['checks']['external_service'] = 'healthy'
        else:
            health_status['checks']['external_service'] = f'unhealthy: HTTP {response.status_code}'
    except Exception as e:
        health_status['checks']['external_service'] = f'unhealthy: {str(e)}'
    
    # Memory usage check
    import psutil
    memory_percent = psutil.virtual_memory().percent
    if memory_percent > 90:
        health_status['checks']['memory'] = f'warning: {memory_percent}% used'
        if health_status['status'] == 'healthy':
            health_status['status'] = 'degraded'
    else:
        health_status['checks']['memory'] = f'healthy: {memory_percent}% used'
    
    status_code = 200 if health_status['status'] == 'healthy' else 503
    return jsonify(health_status), status_code

Rollback Automation

# Automated rollback script
#!/bin/bash

set -e

DEPLOYMENT_ID=$1
THRESHOLD_ERROR_RATE=5
THRESHOLD_RESPONSE_TIME=2000
MONITOR_DURATION=300  # 5 minutes

echo "Monitoring deployment: $DEPLOYMENT_ID"

# Function to get metric value
get_metric_value() {
    local metric_name=$1
    local namespace=$2
    local dimensions=$3
    
    aws cloudwatch get-metric-statistics \
        --namespace "$namespace" \
        --metric-name "$metric_name" \
        --dimensions "$dimensions" \
        --start-time $(date -u -d '5 minutes ago' +%Y-%m-%dT%H:%M:%S) \
        --end-time $(date -u +%Y-%m-%dT%H:%M:%S) \
        --period 300 \
        --statistics Average \
        --query 'Datapoints[0].Average' \
        --output text
}

# Monitor deployment health
for i in $(seq 1 $((MONITOR_DURATION / 30))); do
    echo "Health check iteration $i"
    
    # Get error rate
    error_rate=$(get_metric_value "HTTPCode_Target_5XX_Count" "AWS/ApplicationELB" "Name=TargetGroup,Value=enterprise-tg")
    
    # Get response time
    response_time=$(get_metric_value "TargetResponseTime" "AWS/ApplicationELB" "Name=TargetGroup,Value=enterprise-tg")
    
    echo "Error rate: $error_rate, Response time: $response_time ms"
    
    # Check thresholds
    if (( $(echo "$error_rate > $THRESHOLD_ERROR_RATE" | bc -l) )); then
        echo "ERROR: Error rate ($error_rate) exceeds threshold ($THRESHOLD_ERROR_RATE)"
        echo "Initiating rollback..."
        
        aws deploy stop-deployment \
            --deployment-id "$DEPLOYMENT_ID" \
            --auto-rollback-enabled
        
        exit 1
    fi
    
    if (( $(echo "$response_time > $THRESHOLD_RESPONSE_TIME" | bc -l) )); then
        echo "ERROR: Response time ($response_time ms) exceeds threshold ($THRESHOLD_RESPONSE_TIME ms)"
        echo "Initiating rollback..."
        
        aws deploy stop-deployment \
            --deployment-id "$DEPLOYMENT_ID" \
            --auto-rollback-enabled
        
        exit 1
    fi
    
    sleep 30
done

echo "Deployment health check completed successfully"
